285 void ‪SHA256::processBuffer()
286 {
287  // the input bytes are considered as bits strings, where the first bit is the most significant bit of the byte
288 
289  // - append "1" bit to message
290  // - append "0" bits until message length in bit mod 512 is 448
291  // - append length as 64 bit integer
292 
293  // number of bits
294  size_t paddedLength = ‪m_bufferSize * 8;
295 
296  // plus one bit set to 1 (always appended)
297  paddedLength++;
298 
299  // number of bits must be (numBits % 512) = 448
300  size_t lower11Bits = paddedLength & 511;
301  if (lower11Bits <= 448)
302  paddedLength += 448 - lower11Bits;
303  else
304  paddedLength += 512 + 448 - lower11Bits;
305  // convert from bits to bytes
306  paddedLength /= 8;
307 
308  // only needed if additional data flows over into a second block
309  unsigned char extra[‪BlockSize];
310 
311  // append a "1" bit, 128 => binary 10000000
312  if (‪m_bufferSize < ‪BlockSize)
313  ‪m_buffer[‪m_bufferSize] = 128;
314  else
315  extra[0] = 128;

319  ‪m_buffer[i] = 0;
320  for (; i < paddedLength; i++)
